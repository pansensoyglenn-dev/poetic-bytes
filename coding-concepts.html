🔤 Variables & Data Types

📦 Variables and Assignment

Python variables are dynamically typed – you don't need to declare types explicitly. Assignment uses the = operator.

# Variable assignment
name = "Glenn"
age = 28
height = 1.75
is_student = True

# Multiple assignment
x, y, z = 10, 20, 30
print(x, y, z)

# Swapping values
a, b = 5, 10
a, b = b, a
print(a, b)

# Dynamic typing
value = 42
print(type(value))
value = "Hello"
print(type(value))
💡 Python is dynamically typed – variables can change type during runtime. Use type() to check the current type.

🔁 Loops – for and while

Python's for loop iterates over iterables. while loops run while a condition is true.

🎯 Match Statement (Python 3.10+)

Structural pattern matching – like a powerful switch statement.

# Basic match

match value:
    case 0:
        print("Zero")
    case 1:
        print("One")
    case 42:
        print("The answer")
    case _:
        print("Other")

    case n if n > 0:
        print("Positive")
    case n if n < 0:
        print("Negative")
    case _:
        print("Zero")

match point:
    case (0, 0):
        print("Origin")
    case (x, 0):
        print(f"On X-axis at {x}")
    case (0, y):
        print(f"On Y-axis at {y}")
    case (x, y):
        print(f"Point at ({x}, {y})")

🔄 Closures

A closure is a function that remembers its enclosing scope even when called outside that scope.

def make_multiplier(factor):
    def multiplier(x):
        return x * factor
    return multiplier

double = make_multiplier(2)
triple = make_multiplier(3)

print(double(5))
print(triple(5))

# Closure with nonlocal
def counter():
    count = 0
    def increment():
        nonlocal count
        count += 1
        return count
    return increment

c = counter()
print(c())
print(c())
print(c())
💡 Closures are powerful for data hiding and function factories. The nonlocal keyword allows modifying variables from the enclosing scope.
🏗️ Object-Oriented Programming

🏛️ Classes and Objects

Classes are blueprints for creating objects. They encapsulate data (attributes) and behaviour (methods).

class Car:
    def __init__(self, make, model, year):
        self.make = make
        self.model = model
        self.year = year
        self.odometer = 0

    def drive(self, miles):

car = Car("Toyota", "Camry", 2023)

🔒 Encapsulation and Properties

Use @property to control access to attributes, enabling getters and setters.

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self._balance = balance

    @property
    def balance(self):
        return self._balance

    @balance.setter
    def balance(self, value):
        if value < 0:
            raise ValueError("Balance cannot be negative")
        self._balance = value

    def deposit(self, amount):
        if amount <= 0:
            raise ValueError("Deposit amount must be positive")
        self._balance += amount

    def withdraw(self, amount):
        if amount > self._balance:
            raise ValueError("Insufficient funds")
        self._balance -= amount

acc = BankAccount("Glenn", 1000)
print(acc.balance)
acc.deposit(500)
print(acc.balance)

🌿 Inheritance

Inheritance allows a class to extend and reuse behaviour from a parent class.

class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        raise NotImplementedError("Subclass must implement speak()")

    def __str__(self):
        return f"Animal: {self.name}"

class Dog(Animal):
    def speak(self):
        return f"{self.name} says Woof!"

class Cat(Animal):
    def speak(self):
        return f"{self.name} says Meow!"

dog = Dog("Rex")
cat = Cat("Whiskers")

print(dog.speak())
print(cat.speak())

# Using super() to call parent method
class GoldenRetriever(Dog):
    def speak(self):
        parent_speak = super().speak()
        return f"{parent_speak} And I'm a Golden Retriever!"

golden = GoldenRetriever("Buddy")
print(golden.speak())

🎭 Polymorphism

Polymorphism allows objects of different classes to be treated uniformly through a common interface.

# Duck typing
def make_sound(animal):
    return animal.speak()

print(make_sound(dog))
print(make_sound(cat))

# Abstract Base Classes
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

    @abstractmethod
    def perimeter(self):
        pass

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height

    def area(self):
        return self.width * self.height

    def perimeter(self):
        return 2 * (self.width + self.height)

rect = Rectangle(5, 3)
print(rect.area())
print(rect.perimeter())
⚡ Advanced Python

🎨 Decorators

Decorators are functions that modify or extend the behaviour of other functions.

import time
from functools import wraps

# Basic decorator
def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end - start:.4f} seconds")
        return result
    return wrapper

# Decorator with parameters
def retry(times=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            attempts = 0
            while attempts < times:
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    attempts += 1
                    if attempts == times:
                        raise
                    time.sleep(delay)
        return wrapper
    return decorator

# Using the decorators
@timer
def slow_function():
    time.sleep(1)
    return "Done"

@retry(times=3, delay=0.5)
def unreliable_function():
    import random
    if random.random() < 0.7:
        raise ValueError("Random failure")
    return "Success!"

print(slow_function())
print(unreliable_function())

⚡ Generators and Iterators

Generators produce values lazily using yield, saving memory for large datasets.

# Generator function
def fibonacci(limit):
    a, b = 0, 1
    while a < limit:
        yield a
        a, b = b, a + b

for num in fibonacci(100):
    print(num, end=" ")

# Generator expression
squares = (x**2 for x in range(10))
for sq in squares:
    print(sq, end=" ")

# Custom iterator
class CountDown:
    def __init__(self, start):
        self.start = start

    def __iter__(self):
        return self

    def __next__(self):
        if self.start < 0:
            raise StopIteration
        current = self.start
        self.start -= 1
        return current

for n in CountDown(5):
    print(n, end=" ")

🔐 Context Managers

Context managers manage resources with with statements, ensuring proper setup and teardown.

# Using built-in context manager
with open("example.txt", "w") as f:
    f.write("Hello, World!")

# Custom context manager using class
class ManagedFile:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode

    def __enter__(self):
        self.file = open(self.filename, self.mode)
        return self.file

    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
        if exc_type is not None:
            print(f"Exception occurred: {exc_val}")
        return False

with ManagedFile("example.txt", "r") as f:
    content = f.read()
    print(content)

# Context manager using contextlib
from contextlib import contextmanager

@contextmanager
def managed_resource(resource):
    print("Acquiring resource")
    yield resource
    print("Releasing resource")

with managed_resource("database") as res:
    print(f"Using {res}")

🚀 Async/Await

Async functions allow non-blocking I/O operations, improving performance for network-heavy workloads.

import asyncio

# Basic async function
async def say_hello(delay, name):
    await asyncio.sleep(delay)
    return f"Hello, {name}!"

# Running async functions
async def main():
    result = await say_hello(1, "Glenn")
    print(result)

asyncio.run(main())

# Concurrent tasks
async def fetch_all():
    tasks = [
        say_hello(2, "Alice"),
        say_hello(1, "Bob"),
        say_hello(3, "Charlie")
    ]
    results = await asyncio.gather(*tasks)
    return results

results = asyncio.run(fetch_all())
print(results)

# TaskGroup (Python 3.11+)
async def main_with_group():
    async with asyncio.TaskGroup() as tg:
        task1 = tg.create_task(say_hello(1, "Task 1"))
        task2 = tg.create_task(say_hello(2, "Task 2"))
    print(task1.result())
    print(task2.result())

# asyncio.run(main_with_group())
💡 Async is ideal for I/O-bound tasks like network requests, database queries, and file operations. For CPU-bound tasks, use multiprocessing instead.

🧮 Recursion

Recursive functions call themselves to solve problems that have smaller subproblems.

# Factorial recursively
def factorial(n: int) -> int:
    if n <= 1:
        return 1
    return n * factorial(n - 1)

print(factorial(5))

# Fibonacci recursively (inefficient without memoization)
def fibonacci(n: int) -> int:
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))

# Recursion with memoization (optimized)
from functools import lru_cache

@lru_cache(maxsize=None)
def fib_memo(n: int) -> int:
    if n <= 1:
        return n
    return fib_memo(n - 1) + fib_memo(n - 2)

print(fib_memo(50))

# Recursive directory traversal
import os

def list_files(path, indent=0):
    for item in os.listdir(path):
        full_path = os.path.join(path, item)
        print("  " * indent + item)
        if os.path.isdir(full_path):
            list_files(full_path, indent + 1)

🧵 Threading vs Multiprocessing

Threading for I/O-bound tasks, multiprocessing for CPU-bound tasks.

from threading import Thread
from multiprocessing import Process
import time

# Threading (I/O-bound)
def io_task(name, delay):
    time.sleep(delay)
    return f"Task {name} finished"

threads = []
for i in range(5):
    t = Thread(target=io_task, args=(i, 1))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

# Multiprocessing (CPU-bound)
def cpu_task(n):
    return sum(i * i for i in range(n))

from concurrent.futures import ProcessPoolExecutor

with ProcessPoolExecutor(max_workers=4) as executor:
    results = list(executor.map(cpu_task, [1000000, 1000000, 1000000]))

# When to use what
# - I/O-bound: threading, asyncio
# - CPU-bound: multiprocessing
💡 threading works well for I/O-bound tasks because the GIL is released during I/O operations. For CPU-bound tasks, use multiprocessing to bypass the GIL.

📝 Decorators with Arguments (Decorator Factories)

Decorator factories return decorators that accept parameters.

from functools import wraps
import time
import logging

def log(level=logging.INFO):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            logging.log(level, f"Calling {func.__name__}")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@log(level=logging.DEBUG)
def process_data(data):
    return data * 2

# Class decorator
def add_method(func):
    def decorator(cls):
        setattr(cls, func.__name__, func)
        return cls
    return decorator

@add_method
def hello(self):
    return "Hello from the class!"

@add_method(hello)
class MyClass:
    pass

obj = MyClass()
print(obj.hello())

🔮 Metaclasses (Advanced)

Metaclasses are classes of classes. They define how classes behave.

# Simple metaclass
class Meta(type):
    def __new__(cls, name, bases, dct):
        dct["version"] = "1.0"
        return super().__new__(cls, name, bases, dct)

    def __call__(cls, *args, **kwargs):
        instance = super().__call__(*args, **kwargs)
        instance.created_at = time.time()
        return instance

class MyClass(metaclass=Meta):
    def __init__(self, name):
        self.name = name

obj = MyClass("Glenn")
print(obj.version)
print(obj.created_at)

# Singleton metaclass
class Singleton(type):
    _instances = {}
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=Singleton):
    def __init__(self):
        self.connected = False

db1 = Database()
db2 = Database()
print(db1 is db2)
💡 Metaclasses are powerful but rarely needed in everyday Python. They're used in frameworks like Django and SQLAlchemy.

import re

text = "Contact me at glenn@example.com or call +1-555-1234"

emails = re.findall(email_pattern, text)

    print(match.group("year"))
    print(match.group("month"))
    print(match.group("day"))

"

", html)

📝 Context Managers with contextlib

The contextlib module provides utilities for creating context managers.

from contextlib import contextmanager, suppress, closing
import urllib.request

# suppress context manager
with suppress(FileNotFoundError):
    with open("nonexistent.txt") as f:
        content = f.read()

# closing context manager
with closing(urllib.request.urlopen("https://example.com")) as response:
    data = response.read()

# Custom context manager with contextmanager
@contextmanager
def timed(name):
    print(f"Starting {name}")
    start = time.time()
    try:
        yield
    finally:
        end = time.time()
        print(f"{name} took {end - start:.2f} seconds")

with timed("sleep"):
    time.sleep(1)

# ExitStack for managing multiple context managers
from contextlib import ExitStack

with ExitStack() as stack:
    files = [
        stack.enter_context(open("file1.txt", "w")),
        stack.enter_context(open("file2.txt", "w"))
    ]
    for f in files:
        f.write("Hello")

🧬 Descriptors

Descriptors control attribute access in classes. They power @property and @classmethod.

# Data descriptor
class PositiveNumber:
    def __set_name__(self, owner, name):
        self.name = name

    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return obj.__dict__.get(self.name, 0)

    def __set__(self, obj, value):
        if value < 0:
            raise ValueError(f"{self.name} must be positive")
        obj.__dict__[self.name] = value

class Product:
    price = PositiveNumber()
    quantity = PositiveNumber()

    def __init__(self, price, quantity):
        self.price = price
        self.quantity = quantity

p = Product(10, 5)
print(p.price)
# p.price = -5  # Raises ValueError

# Non-data descriptor (only __get__)
class LazyAttribute:
    def __init__(self, func):
        self.func = func

    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        value = self.func(obj)
        setattr(obj, self.func.__name__, value)
        return value

class Circle:
    def __init__(self, radius):
        self.radius = radius

    @LazyAttribute
    def area(self):
        print("Calculating area...")
        return 3.14159 * self.radius ** 2

c = Circle(5)
print(c.area)
print(c.area)

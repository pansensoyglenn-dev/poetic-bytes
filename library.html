Python Standard Library

Most useful modules • with short examples

datetime

core
  • datetime – dates, times, timezones
Example
from datetime import datetime, timedelta, timezone

now = datetime.now()
print(now.strftime("%Y-%m-%d %H:%M"))

future = now + timedelta(days=30)
print(future)

utc = datetime.now(timezone.utc)
print(utc)

Text processing

  • re • string • textwrap
Example
import re
from textwrap import dedent

text = "  Hello   world!  "
clean = re.sub(r'\s+', ' ', text.strip())
print(clean)

print(dedent("""
    This is a
    multi-line string
    that stays aligned.
"""))

Data formats

  • json • csv • pickle
Example
import json
import csv
from io import StringIO

data = {"name": "Alice", "score": 95}
print(json.dumps(data, indent=2))

csv_str = StringIO()
writer = csv.DictWriter(csv_str, fieldnames=["name","score"])
writer.writeheader()
writer.writerow(data)
print(csv_str.getvalue())

Concurrency

  • threading • asyncio • concurrent.futures
Example
import asyncio
from concurrent.futures import ThreadPoolExecutor

async def fetch(url):
    await asyncio.sleep(1)
    return f"result from {url}"

async def main():
    async with asyncio.TaskGroup() as tg:
        tasks = [tg.create_task(fetch(f"https://site{i}.com")) for i in range(3)]
    print([t.result() for t in tasks])

asyncio.run(main())

Networking

  • urllib • http.client • socket
Example
from urllib.request import urlopen
from urllib.parse import urlencode, quote

query = urlencode({"q": "python programming"})
url = f"https://duckduckgo.com/html/?{query}"
with urlopen(url) as resp:
    print(resp.read(500).decode()[:300])

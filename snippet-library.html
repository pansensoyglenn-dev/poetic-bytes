Decorators

Functions that modify other functions, with timing example and proper metadata preservation using functools.wraps.

import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end - start:.4f} seconds")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "Done"

class BankAccount:

Context Managers

Resource management with the with statement, implementing __enter__ and __exit__.

class ManagedFile:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
    def __enter__(self):
        self.file = open(self.filename, self.mode)
        return self.file
    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()

with ManagedFile("example.txt", "w") as f:
    f.write("Hello, World!")

Binary Search

Efficient O(log n) search algorithm for sorted arrays.

def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [1, 3, 5, 7, 9, 11, 13]
print(binary_search(arr, 7))
print(binary_search(arr, 4))

while left <= right:Continues while search space is valid

if arr[mid] == target:Found the target → returns index
elif arr[mid] < target:Target is in right half → moves left boundary
else:Target is in left half → moves right boundary
return -1Target not found in array

Quicksort

Divide and conquer sorting algorithm with O(n log n) average time complexity.

def quicksort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quicksort(left) + middle + quicksort(right)

data = [3, 6, 8, 10, 1, 2, 1]
print(quicksort(data))

Singleton Pattern

Design pattern that ensures only one instance of a class exists.

class SingletonMeta(type):
    _instances = {}
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Config(metaclass=SingletonMeta):
    def __init__(self):
        self.settings = {"debug": True}

c1 = Config()
c2 = Config()
print(c1 is c2)

class SingletonMeta(type):Metaclass for Singleton pattern
_instances = {}Dictionary to store instances
def __call__(cls, *args, **kwargs):Intercepts instance creation
if cls not in cls._instances:Checks if instance already exists

Singly Linked List

Implementation with traversal and insertion using Node and LinkedList classes.

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        current = self.head
        while current.next:
            current = current.next
        current.next = new_node

    def display(self):
        elements = []
        current = self.head
        while current:
            elements.append(current.data)
            current = current.next
        return elements

ll = LinkedList()
for i in range(1, 6):
    ll.append(i)
print(ll.display())

Stack (LIFO) with List

Stack implementation using a list with push, pop, and peek operations.

class Stack:
    def __init__(self):
        self.items = []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        if not self.is_empty():
            return self.items.pop()

    def peek(self):
        if not self.is_empty():
            return self.items[-1]

    def is_empty(self):
        return len(self.items) == 0

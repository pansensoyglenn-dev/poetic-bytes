import os, shutil
from pathlib import Path

def organize_files(folder):
    extensions = {
        "images": [".jpg", ".png", ".gif"],
        "documents": [".pdf", ".docx", ".txt"],
        "spreadsheets": [".xlsx", ".csv"],
        "archives": [".zip", ".tar", ".gz"]
    }
    for file in Path(folder).iterdir():
        if file.is_file():
            for folder_name, exts in extensions.items():
                if file.suffix.lower() in exts:
                    dest = Path(folder) / folder_name
                    dest.mkdir(exist_ok=True)
                    shutil.move(str(file), str(dest / file.name))
                    print(f"Moved {file.name} to {folder_name}")
                    break

File System Automation

Working with files and folders using os, shutil, and pathlib

📂 Working with Paths using pathlib

pathlib is the modern, object-oriented way to handle file paths. It's cleaner and more cross‑platform than os.path.

from pathlib import Path

# Current directory
current = Path.cwd()
print("Current:", current)

# Home directory
home = Path.home()
print("Home:", home)

# Create a path
file_path = Path("documents") / "report.txt"
print("File path:", file_path)

# Check if exists
print("Exists:", file_path.exists())
print("Is file:", file_path.is_file())
print("Is dir:", file_path.is_dir())

# File properties
print("Name:", file_path.name)
print("Stem:", file_path.stem)
print("Suffix:", file_path.suffix)
print("Parent:", file_path.parent)

📋 Listing and Filtering Files

List files in directories with wildcards using .glob() and .rglob().

from pathlib import Path

# List all .py files in current directory
for py_file in Path(".").glob("*.py"):
    print(py_file)

# List all .txt files recursively
for txt_file in Path(".").rglob("*.txt"):
    print(txt_file)

# Filter by size
large_files = [f for f in Path(".").iterdir() if f.is_file() and f.stat().st_size > 1_000_000]
print(f"Found {len(large_files)} large files")

Use shutil for high‑level file operations — copy, move, delete, and archive.

🔄 File Organizer — Real‑World Automation

Automatically sort files into folders by extension — a script you'll actually use.

import shutil
from pathlib import Path

def organize_by_extension(folder):
    folder = Path(folder)
    for file in folder.iterdir():
        if file.is_file():
            ext = file.suffix[1:].lower() or "no_extension"
            dest = folder / ext
            dest.mkdir(exist_ok=True)
            shutil.move(str(file), str(dest / file.name))
            print(f"Moved {file.name} → {ext}/")

# Usage
organize_by_extension("Downloads")
Moved photo.jpg → jpg/
Moved data.xlsx → xlsx/
Moved report.pdf → pdf/
Moved script.py → py/
💡 This script is a real time‑saver — run it on your Downloads folder every week to keep it clean.

Web Scraping

Extracting data from websites with requests, BeautifulSoup, and Selenium

🌐 Basic Scraping with Requests & BeautifulSoup

Send HTTP requests and parse HTML to extract the data you need.

import requests
from bs4 import BeautifulSoup

# Fetch a page
url = "https://example.com"
response = requests.get(url)
response.raise_for_status()  # Raise error for bad status

# Parse HTML
soup = BeautifulSoup(response.text, "html.parser")

# Find elements
title = soup.find("h1")
print("Title:", title.text if title else "None")

# Find all links
links = soup.find_all("a")
for link in links[:5]:
    print(link.get("href"))

# Find by class
items = soup.find_all("div", class_="item")
for item in items:
    print(item.text.strip())

💡 Always respect robots.txt and add a User-Agent header to identify your scraper. Never overload a site with requests.

🕷️ Scraping with Selenium — JavaScript‑Rendered Pages

Selenium controls a real browser, so it can handle pages that load content dynamically with JavaScript.

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Setup (requires chromedriver)
driver = webdriver.Chrome()
driver.get("https://example.com")

# Wait for element to appear
element = WebDriverWait(driver, 10).until(
    EC.presence_of_element_located((By.ID, "content"))
)

# Find elements
items = driver.find_elements(By.CLASS_NAME, "item")
for item in items:
    print(item.text)

# Click a button
button = driver.find_element(By.ID, "load-more")
button.click()

# Take a screenshot
driver.save_screenshot("screenshot.png")

driver.quit()
[Selenium: browser automation with waiting and clicking]
💡 Selenium is heavier than requests but essential for JavaScript-heavy sites. Use headless mode for servers: options.add_argument('--headless').

📊 Scraping with Playwright — Modern Alternative

Playwright is a newer, more powerful alternative to Selenium with better API design and faster execution.

from playwright.sync_api import sync_playwright

with sync_playwright() as p:
    browser = p.chromium.launch()
    page = browser.new_page()
    page.goto("https://example.com")
    
    # Wait for and click

    # Extract data

    for item in items:
        print(item.inner_text())
    
    browser.close()

Task Scheduling

Running scripts automatically with schedule, APScheduler, and cron

⏰ Simple Scheduling with the schedule Library

Lightweight and easy — schedule functions to run at intervals or specific times.

import schedule
import time

def daily_backup():
    print("Running daily backup...")

def send_report():
    print("Sending weekly report...")

# Schedule tasks
schedule.every().day.at("02:00").do(daily_backup)
schedule.every().monday.at("09:00").do(send_report)
schedule.every(30).minutes.do(print, "30 minutes passed")
schedule.every().hour.do(print, "Hourly check")

# Keep running
while True:
    schedule.run_pending()
    time.sleep(1)
[Scheduler running: daily at 02:00, weekly Monday 09:00, every 30 min, every hour]
💡 schedule is great for simple scripts. For production, use APScheduler or system cron.

⚙️ APScheduler — Advanced Scheduling

More powerful — supports cron‑style expressions, persistent jobs, and concurrency control.

from apscheduler.schedulers.blocking import BlockingScheduler
from apscheduler.triggers.cron import CronTrigger

scheduler = BlockingScheduler()

@scheduler.scheduled_job("interval", minutes=5)
def job_1():
    print("Runs every 5 minutes")

@scheduler.scheduled_job(CronTrigger(day_of_week="mon", hour=9))
def job_2():
    print("Runs every Monday at 9 AM")

# Start the scheduler
try:
    scheduler.start()
except (KeyboardInterrupt, SystemExit):
    scheduler.shutdown()
[APScheduler running: interval 5 min, cron Monday 9 AM]
💡 APScheduler supports 'interval', 'cron', and 'date' triggers. It's the standard for production Python apps.

🐧 System Cron — Linux / macOS

For reliable, system‑wide scheduling, use cron — the Unix task scheduler.

# Edit crontab
crontab -e

# Run every hour
0 * * * * /home/user/hourly_task.py
[Cron: daily 2 AM, weekly Monday 9 AM, every 30 min, every hour]
💡 Cron is the most reliable method for long‑running servers. Use absolute paths for Python scripts and log output to a file.

Email Automation

Sending and receiving emails with smtplib, email, and imaplib

📧 Sending Emails with smtplib

Send automated emails from your Python scripts — alerts, reports, and notifications.

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

def send_email(subject, body, to_email):
    from_email = "your_email@gmail.com"
    password = "your_app_password"
    
    msg = MIMEMultipart()
    msg["From"] = from_email
    msg["To"] = to_email
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "plain"))
    
    try:
        server = smtplib.SMTP("smtp.gmail.com", 587)
        server.starttls()
        server.login(from_email, password)
        server.send_message(msg)
        print("Email sent!")
    except Exception as e:
        print(f"Error: {e}")
    finally:
        server.quit()

send_email("Daily Report", "Here is today's report...", "recipient@example.com")
[Email sent: Daily Report to recipient@example.com]
💡 For Gmail, use an App Password (not your regular password) — enable 2‑Factor Authentication first. Never hardcode credentials; use environment variables.

📥 Reading Emails with IMAP

Check your inbox programmatically — process attachments, parse content, and trigger actions.

import imaplib
import email
from email.policy import default

def read_inbox(email_address, password):
    mail = imaplib.IMAP4_SSL("imap.gmail.com")
    mail.login(email_address, password)
    mail.select("inbox")
    
    # Search for unread emails
    status, messages = mail.search(None, "UNSEEN")
    for num in messages[0].split():
        status, data = mail.fetch(num, "(RFC822)")
        msg = email.message_from_bytes(data[0][1], policy=default)
        
        subject = msg["Subject"]
        from_addr = msg["From"]
        print(f"From: {from_addr}")
        print(f"Subject: {subject}")
        
        # Extract body
        if msg.is_multipart():
            for part in msg.walk():
                if part.get_content_type() == "text/plain":
                    body = part.get_content()
                    print(f"Body: {body[:100]}...")
        
        mail.store(num, "+FLAGS", "\\SEEN")  # Mark as read
    
    mail.close()
    mail.logout()

read_inbox("your_email@gmail.com", "your_password")
[Reading inbox: unread emails parsed successfully]
💡 IMAP is great for building email‑triggered workflows — process attachments, parse commands, and take actions automatically.

Working with spreadsheets using openpyxl, pandas, and PDFs with PyPDF2

📊 Excel Automation with openpyxl

Read, write, and modify Excel files — create reports, analyze data, and automate spreadsheet tasks.

from openpyxl import Workbook, load_workbook
from openpyxl.styles import Font, PatternFill

# Create a new workbook
wb = Workbook()
ws = wb.active
ws.title = "Sales Data"

# Add headers
headers = ["Product", "Q1", "Q2", "Q3", "Q4"]
for col, header in enumerate(headers, start=1):
    cell = ws.cell(row=1, column=col, value=header)
    cell.font = Font(bold=True)

# Add data
data = [
    ["Laptops", 150, 120, 180, 200],
    ["Phones", 300, 280, 350, 400],
    ["Tablets", 80, 90, 75, 110],
]
for row, row_data in enumerate(data, start=2):
    for col, value in enumerate(row_data, start=1):
        ws.cell(row=row, column=col, value=value)

# Add total row
total_row = len(data) + 2
ws.cell(row=total_row, column=1, value="TOTAL")
for col in range(2, 6):
    ws.cell(row=total_row, column=col, value=f"=SUM({chr(64+col)}2:{chr(64+col)}{total_row-1})")

# Save
wb.save("sales_report.xlsx")

# Read an existing file
wb = load_workbook("sales_report.xlsx")
ws = wb.active
for row in ws.iter_rows(values_only=True):
    print(row)
[Excel file created: sales_report.xlsx with headers, data, totals]
💡 openpyxl supports charts, formulas, styling, and large files. For data analysis, use pandas with pd.read_excel().

📄 PDF Processing with PyPDF2

Merge, split, extract text, and rotate PDF pages — automate document workflows.

from PyPDF2 import PdfReader, PdfWriter

# Extract text from PDF
reader = PdfReader("document.pdf")
text = ""
for page in reader.pages:
    text += page.extract_text()
print(text[:500])  # First 500 characters

# Merge PDFs
writer = PdfWriter()
for pdf_file in ["file1.pdf", "file2.pdf"]:
    reader = PdfReader(pdf_file)
    for page in reader.pages:
        writer.add_page(page)
with open("merged.pdf", "wb") as f:
    writer.write(f)

# Split PDF by page
reader = PdfReader("large.pdf")
for i, page in enumerate(reader.pages):
    writer = PdfWriter()
    writer.add_page(page)
    with open(f"page_{i+1}.pdf", "wb") as f:
        writer.write(f)
[PDF processed: text extracted, files merged, pages split]
💡 For more advanced PDF editing, use reportlab for creating PDFs or pdfplumber for extracting tables.

Interacting with REST APIs using requests — fetch, process, and automate data flows

🌐 Working with REST APIs

Automate data retrieval, updates, and integrations with external services.

import requests
import json

# GET request
response = requests.get("https://api.github.com/users/octocat")
response.raise_for_status()
user = response.json()

🔑 API Authentication — Bearer Tokens

Most APIs require authentication. Use Bearer tokens or API keys in headers.

import os
import requests

# Use environment variables for secrets
API_TOKEN = os.environ.get("API_TOKEN", "your-token-here")

headers = {
    "Authorization": f"Bearer {API_TOKEN}",
    "Content-Type": "application/json"
}

response = requests.get(
    "https://api.example.com/protected",
    headers=headers
)
response.raise_for_status()

# For API key authentication
params = {"api_key": API_TOKEN}
response = requests.get(
    "https://api.example.com/data",
    params=params
)
[API authentication with Bearer token and API key]
💡 Never commit API keys to code. Use python-dotenv to load them from a .env file.

📊 Automating Data Pipelines

Fetch data from APIs, process it with pandas, and save to files or databases.

import requests
import pandas as pd

def fetch_and_save_data(endpoint, filename):
    response = requests.get(endpoint)
    response.raise_for_status()
    data = response.json()
    
    # Convert to DataFrame
    df = pd.DataFrame(data)
    
    # Clean and transform
    if "created_at" in df.columns:
        df["created_at"] = pd.to_datetime(df["created_at"])
    
    # Save to CSV
    df.to_csv(filename, index=False)
    print(f"Saved {len(df)} records to {filename}")

# Example: fetch GitHub repos
fetch_and_save_data(
    "https://api.github.com/search/repositories?q=python+stars:>10000",
    "top_python_repos.csv"
)
Saved 30 records to top_python_repos.csv
💡 This pattern is the foundation of data engineering — fetch, transform, and save automatically on a schedule.

Automation FAQ

Quick answers to common automation questions

What can I automate with Python?

+

Almost anything: file organization, web scraping, email sending, report generation, Excel/PDF processing, GUI automation, API data fetching, social media posting, and more.

Is web scraping legal?

+

It depends. Always check the website's robots.txt and terms of service. Scraping public data for personal use is usually fine. Avoid overloading servers — add delays between requests.

What's the best scheduling library?

+

For simple scripts, use schedule. For production systems, use APScheduler or system cron. For cloud deployments, use platform schedulers (AWS CloudWatch, Render cron jobs).

How do I keep automation scripts running 24/7?

+

Deploy to a server or cloud platform (Render, Railway, AWS EC2). Use tmux or screen on Linux, or run as a system service with systemd.

What about GUI automation on different operating systems?

+

pyautogui works on Windows, macOS, and Linux. For Windows‑specific automation, consider pywin32. On macOS, pyobjc provides more control.

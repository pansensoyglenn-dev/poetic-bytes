Auto-Organise a Download Folder

Sort files by type into subfolders — images, docs, videos, code — automatically.

pathlibshutilorganize
Beginner+
from pathlib import Path
import shutil

RULES = {
    "Images":    [".jpg",".jpeg",".png",".gif",".webp",".svg",".heic"],
    "Documents": [".pdf",".doc",".docx",".txt",".md",".odt",".rtf"],
    "Videos":    [".mp4",".mov",".avi",".mkv",".webm"],
    "Audio":     [".mp3",".wav",".flac",".aac",".m4a"],
    "Spreadsheets":[".xls",".xlsx",".csv",".ods"],
    "Code":      [".py",".js",".ts",".html",".css",".json",".sql"],
    "Archives":  [".zip",".tar",".gz",".rar",".7z"],
    "Executables":[".exe",".msi",".dmg",".deb",".pkg"],
}

EXT_TO_FOLDER = {ext: folder for folder, exts in RULES.items() for ext in exts}

def organise(source: str, dry_run: bool = True) -> dict:
    src = Path(source)
    moved = {}
    skipped = []

    for f in src.iterdir():
        if not f.is_file(): continue
        folder = EXT_TO_FOLDER.get(f.suffix.lower(), "Other")
        dest_dir = src / folder
        dest     = dest_dir / f.name

        if dest.exists():
            skipped.append(f.name); continue

        if not dry_run:
            dest_dir.mkdir(exist_ok=True)
            shutil.move(str(f), str(dest))

        moved.setdefault(folder, []).append(f.name)

    print(f"{'[DRY RUN] ' if dry_run else ''}Organising: {source}")
    for folder, files in sorted(moved.items()):
        print(f"  📁 {folder}/ ← {len(files)} files")
        for name in files[:3]:
            print(f"     • {name}")
        if len(files) > 3: print(f"     … and {len(files)-3} more")
    if skipped: print(f"  ⚠️  Skipped (already exist): {len(skipped)} files")
    return moved

# Usage: organise("/Users/you/Downloads", dry_run=True)  → preview
# organise("/Users/you/Downloads", dry_run=False) → actually move
organise(".", dry_run=True)
💡 Always run with dry_run=True first to preview what will move. Set dry_run=False to execute. Adding this to a cron job or Windows Task Scheduler makes your Downloads folder self-cleaning.

Batch Rename Files

Rename hundreds of files at once — add prefixes, clean names, replace text, number sequentially.

pathlibrerenamebatch
Beginner+
import re
from pathlib import Path

def batch_rename(
    folder: str,
    pattern: str = "*",
    prefix: str = "",
    suffix: str = "",
    replace_from: str = "",
    replace_to: str = "",
    sequential: bool = False,
    start: int = 1,
    pad: int = 3,
    dry_run: bool = True,
) -> list[tuple[str, str]]:
    p = Path(folder)
    files = sorted(f for f in p.glob(pattern) if f.is_file())
    renames = []

    for i, f in enumerate(files, start):
        stem = f.stem
        if replace_from: stem = stem.replace(replace_from, replace_to)
        stem = re.sub(r"\s+", "_", stem.strip())     # spaces → underscores
        stem = re.sub(r"[^\w\-.]", "", stem)          # remove special chars
        if sequential: stem = f"{str(i).zfill(pad)}"
        new_name = f"{prefix}{stem}{suffix}{f.suffix.lower()}"
        new_path = f.parent / new_name

        if new_name == f.name: continue
        renames.append((f.name, new_name))

        if not dry_run and not new_path.exists():
            f.rename(new_path)

    tag = "[DRY RUN]" if dry_run else "[RENAMED]"
    print(f"{tag} {len(renames)} files in '{folder}'")
    for old, new in renames[:10]:
        print(f"  {old}  →  {new}")
    if len(renames) > 10: print(f"  … and {len(renames)-10} more")
    return renames

# Clean all .jpg files: lowercase, no spaces, add photo_ prefix
batch_rename(".", pattern="*.txt", prefix="doc_", dry_run=True)

# Rename all photos sequentially: 001.jpg, 002.jpg, …
batch_rename(".", pattern="*.txt", sequential=True, pad=3, dry_run=True)
💡 Use dry_run=True (default) to preview all renames before committing. The preview shows old → new for every file so nothing is a surprise.

Find Duplicate Files

Scan a directory tree, hash every file, and report duplicates by content — not just name.

hashlibpathlibduplicate detection
Intermediate+
import hashlib
from pathlib import Path
from collections import defaultdict

def file_hash(path: Path, chunk: int = 65536) -> str:
    h = hashlib.md5()
    with open(path, "rb") as f:
        while data := f.read(chunk):
            h.update(data)
    return h.hexdigest()

def find_duplicates(root: str, min_size: int = 1024) -> dict[str, list[Path]]:
    root_path = Path(root)
    by_size: dict[int, list[Path]] = defaultdict(list)
    print(f"Scanning {root}…")

    for f in root_path.rglob("*"):
        if f.is_file() and f.stat().st_size >= min_size:
            by_size[f.stat().st_size].append(f)

    duplicates: dict[str, list[Path]] = defaultdict(list)
    candidates = {sz: paths for sz, paths in by_size.items() if len(paths) > 1}
    print(f"Hashing {sum(len(v) for v in candidates.values())} candidate files…")

    for paths in candidates.values():
        for path in paths:
            try: duplicates[file_hash(path)].append(path)
            except PermissionError: pass

    return {h: p for h, p in duplicates.items() if len(p) > 1}

def report_duplicates(dups: dict):
    if not dups:
        print("✅ No duplicates found!"); return
    total_wasted = 0
    print(f"\n{'─'*55}")
    print(f"  🔍 Found {len(dups)} duplicate groups")
    print(f"{'─'*55}")
    for i, (h, paths) in enumerate(dups.items(), 1):
        size = paths[0].stat().st_size
        wasted = size * (len(paths) - 1)
        total_wasted += wasted
        print(f"\n  Group {i} ({size/1024:.1f} KB each — {len(paths)} copies):")
        for p in paths:
            print(f"    {p}")
    print(f"\n  💾 Reclaimable space: {total_wasted/1_048_576:.2f} MB")
    print("  ℹ️  Review before deleting — keep the path you prefer.")

dups = find_duplicates(".", min_size=100)
report_duplicates(dups)
💡 Uses size as a first-pass filter (much faster than hashing everything), then MD5 for exact comparison. For extremely large files, use partial hashing (first + last 64 KB) to speed up the scan.

Automated Backup with Rotation

Zip a directory, timestamp it, and keep only the N most recent backups.

zipfileshutilbackuprotation
Intermediate+
import zipfile, shutil
from pathlib import Path
from datetime import datetime

def backup(
    source: str,
    dest_dir: str,
    keep: int = 7,
    prefix: str = "backup",
    exclude_patterns: list[str] | None = None,
) -> Path:
    src  = Path(source)
    dest = Path(dest_dir)
    dest.mkdir(parents=True, exist_ok=True)
    excl = exclude_patterns or ["__pycache__", ".git", ".env", "node_modules"]

    ts       = datetime.now().strftime("%Y%m%d_%H%M%S")
    zip_path = dest / f"{prefix}_{ts}.zip"

    def should_exclude(p: Path) -> bool:
        return any(part in excl for part in p.parts)

    added = 0
    with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as zf:
        for f in src.rglob("*"):
            if f.is_file() and not should_exclude(f):
                zf.write(f, f.relative_to(src))
                added += 1

    size_mb = zip_path.stat().st_size / 1_048_576
    print(f"  ✅ Backed up {added} files → {zip_path.name} ({size_mb:.2f} MB)")

    old_backups = sorted(dest.glob(f"{prefix}_*.zip"))[:-keep]
    for old in old_backups:
        old.unlink()
        print(f"  🗑️  Removed old backup: {old.name}")

    remaining = sorted(dest.glob(f"{prefix}_*.zip"))
    print(f"  📦 Keeping {len(remaining)} backups")
    return zip_path

backup(".", "./backups", keep=5, prefix="project")
⚡ Schedule this with cron or Windows Task Scheduler to run nightly. The rotation logic ensures you never accumulate more than keep backups, preventing disk fill-up over time.

def sales_report(csv_data: str) -> None:
    reader = csv.DictReader(io.StringIO(csv_data))
    rows = list(reader)

    by_region  = defaultdict(lambda: {"units":0,"revenue":0})
    by_product = defaultdict(lambda: {"units":0,"revenue":0})
    by_month   = defaultdict(lambda: {"units":0,"revenue":0})

    for r in rows:
        units, rev = int(r["units"]), float(r["revenue"])
        month = r["date"][:7]
        by_region[r["region"]]["units"]    += units
        by_region[r["region"]]["revenue"]  += rev
        by_product[r["product"]]["units"]  += units
        by_product[r["product"]]["revenue"]+= rev
        by_month[month]["units"]            += units
        by_month[month]["revenue"]          += rev

    total_rev   = sum(float(r["revenue"]) for r in rows)
    total_units = sum(int(r["units"]) for r in rows)
    avg_order   = total_rev / len(rows)

    print(f"{'═'*52}")
    print(f"  📊 SALES REPORT")
    print(f"{'─'*52}")
    print(f"  Total Revenue: ₱{total_rev:>10,.2f}")
    print(f"  Total Units:   {total_units:>10,}")
    print(f"  Avg Order:     ₱{avg_order:>10,.2f}")
    print(f"  Transactions:  {len(rows):>10,}")

    print(f"\n  {'Region':<10} {'Units':>8} {'Revenue':>12} {'% of Total':>11}")
    print(f"  {'─'*10} {'─'*8} {'─'*12} {'─'*11}")
    for reg, d in sorted(by_region.items(), key=lambda x: -x[1]["revenue"]):
        pct = d["revenue"] / total_rev * 100
        bar = "█" * int(pct/4)
        print(f"  {reg:<10} {d['units']:>8,} ₱{d['revenue']:>10,.2f} {pct:>9.1f}% {bar}")

    print(f"\n  {'Product':<12} {'Units':>8} {'Revenue':>12} {'Avg/unit':>10}")
    print(f"  {'─'*12} {'─'*8} {'─'*12} {'─'*10}")
    for prod, d in sorted(by_product.items(), key=lambda x: -x[1]["revenue"]):
        avg = d["revenue"] / d["units"] if d["units"] else 0
        print(f"  {prod:<12} {d['units']:>8,} ₱{d['revenue']:>10,.2f} ₱{avg:>8.2f}")

    print(f"\n  Monthly Trend:")
    max_rev = max(d["revenue"] for d in by_month.values())
    for month in sorted(by_month.keys()):
        d   = by_month[month]
        bar = "▓" * int(d["revenue"] / max_rev * 25)
        print(f"  {month}  {bar:<25} ₱{d['revenue']:>9,.2f}")

Merge Multiple CSV Files

Combine many CSV files from a folder into one, with deduplication and source tracking.

csvpathlibmergededup
Beginner+
import csv, io
from pathlib import Path

def merge_csvs(
    folder: str,
    output: str = "merged.csv",
    add_source_col: bool = True,
    deduplicate: bool = True,
) -> int:
    folder_path = Path(folder)
    csv_files   = sorted(folder_path.glob("*.csv"))
    if not csv_files:
        print("No CSV files found"); return 0

    all_rows:  list[dict] = []
    fieldnames: list[str] = []

    for csv_file in csv_files:
        print(f"  Reading {csv_file.name}…")
        with open(csv_file, newline="", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            rows   = list(reader)
            if not fieldnames:
                fieldnames = list(reader.fieldnames or [])
            if add_source_col:
                for row in rows:
                    row["_source"] = csv_file.name
            all_rows.extend(rows)
            print(f"    → {len(rows)} rows")

    if add_source_col and "_source" not in fieldnames:
        fieldnames.append("_source")

    if deduplicate:
        before = len(all_rows)
        seen   = set()
        unique = []
        for row in all_rows:
            key = tuple((k, v) for k, v in row.items() if k != "_source")
            if key not in seen:
                seen.add(key); unique.append(row)
        all_rows = unique
        print(f"  🔍 Removed {before - len(all_rows)} duplicate rows")

    out_path = Path(folder) / output
    with open(out_path, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames, extrasaction="ignore")
        writer.writeheader()
        writer.writerows(all_rows)

    print(f"\n  ✅ Merged {len(csv_files)} files → {out_path.name}")
    print(f"  Total rows: {len(all_rows):,}")
    return len(all_rows)

import time, requests, logging
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

logging.basicConfig(level=logging.INFO, format="%(levelname)s %(message)s")
log = logging.getLogger(__name__)

def make_session(
    max_retries: int = 3,
    backoff: float  = 0.5,
    timeout: int    = 10,
    headers: dict | None = None,
) -> requests.Session:
    session = requests.Session()
    retry   = Retry(
        total         = max_retries,
        backoff_factor= backoff,
        status_forcelist=[429, 500, 502, 503, 504],
        allowed_methods=["GET", "POST"],
    )
    adapter = HTTPAdapter(max_retries=retry)
    session.mount("https://", adapter)
    session.mount("http://", adapter)
    session.headers.update(headers or {
        "User-Agent": "Mozilla/5.0 PoeticBytes-Bot/1.0",
        "Accept":     "application/json, text/html",
    })
    session._timeout = timeout
    return session

def safe_get(session: requests.Session, url: str, rate_limit: float = 0) -> dict | str | None:
    try:
        log.info(f"GET {url}")
        resp = session.get(url, timeout=session._timeout)
        resp.raise_for_status()
        if rate_limit: time.sleep(rate_limit)
        ct = resp.headers.get("content-type", "")
        return resp.json() if "json" in ct else resp.text
    except requests.HTTPError as e:
        log.error(f"HTTP {e.response.status_code}: {url}")
    except requests.ConnectionError:
        log.error(f"Connection failed: {url}")
    except requests.Timeout:
        log.error(f"Timeout: {url}")
    return None

def batch_fetch(urls: list[str], rate_limit: float = 0.5) -> list[dict]:
    session = make_session()
    results = []
    for url in urls:
        data = safe_get(session, url, rate_limit)
        if data: results.append({"url": url, "data": data})
    print(f"Fetched {len(results)}/{len(urls)} URLs successfully")
    return results

URLS = [
    "https://jsonplaceholder.typicode.com/todos/1",
    "https://jsonplaceholder.typicode.com/todos/2",
    "https://jsonplaceholder.typicode.com/users/1",
]
results = batch_fetch(URLS, rate_limit=0.2)
for r in results:
    print(f"  {r['url'].split('/')[-2]}/{r['url'].split('/')[-1]}: {str(r['data'])[:60]}…")
⚠️ Always respect robots.txt and add time.sleep() between requests. Most websites have rate limits — hammering them can get your IP blocked. A delay of 0.5–2 seconds between requests is standard practice.

import csv, io, urllib.request, json

def fetch_all_pages(base_url: str, max_pages: int = 5) -> list[dict]:
    all_data = []
    for page in range(1, max_pages + 1):
        url = f"{base_url}?_page={page}&_limit=5"
        try:
            with urllib.request.urlopen(url, timeout=10) as resp:
                batch = json.loads(resp.read())
            if not batch: break
            all_data.extend(batch)
            print(f"  Page {page}: {len(batch)} records")
        except Exception as e:
            print(f"  Page {page} failed: {e}"); break
    return all_data

def transform(records: list[dict]) -> list[dict]:
    return [
        {
            "id":       r.get("id"),
            "title":    r.get("title", "").title(),
            "user_id":  r.get("userId"),
            "completed": "Yes" if r.get("completed") else "No",
        }
        for r in records
    ]

def to_csv(records: list[dict]) -> str:
    if not records: return ""
    buf = io.StringIO()
    writer = csv.DictWriter(buf, fieldnames=list(records[0].keys()))
    writer.writeheader()
    writer.writerows(records)
    return buf.getvalue()

print("Fetching paginated data…")
raw      = fetch_all_pages("https://jsonplaceholder.typicode.com/todos", max_pages=3)
cleaned  = transform(raw)
csv_data = to_csv(cleaned)
print(f"\n✅ Fetched {len(cleaned)} records")
print(f"\nCSV preview (first 3 rows):")
print("\n".join(csv_data.split("\n")[:4]))

Send HTML Email with Attachment

Compose and send a professional HTML email with file attachment using stdlib only.

import smtplib, os
from email.mime.multipart import MIMEMultipart
from email.mime.text      import MIMEText
from email.mime.base      import MIMEBase
from email               import encoders
from pathlib              import Path

def send_email(
    to:          str | list[str],
    subject:     str,
    body_html:   str,
    body_text:   str = "",
    attachments: list[str] | None = None,
    smtp_host:   str = "smtp.gmail.com",
    smtp_port:   int = 587,
    sender:      str = os.environ.get("EMAIL_SENDER", "you@gmail.com"),
    password:    str = os.environ.get("EMAIL_PASSWORD", ""),
) -> bool:
    recipients = [to] if isinstance(to, str) else to
    msg = MIMEMultipart("mixed")
    msg["From"]    = sender
    msg["To"]      = ", ".join(recipients)
    msg["Subject"] = subject

    alt = MIMEMultipart("alternative")
    if body_text: alt.attach(MIMEText(body_text, "plain", "utf-8"))
    alt.attach(MIMEText(body_html, "html", "utf-8"))
    msg.attach(alt)

    for path_str in (attachments or []):
        p = Path(path_str)
        if not p.exists(): continue
        part = MIMEBase("application", "octet-stream")
        part.set_payload(p.read_bytes())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", f'attachment; filename="{p.name}"')
        msg.attach(part)

    try:
        with smtplib.SMTP(smtp_host, smtp_port) as server:
            server.ehlo()
            server.starttls()
            server.login(sender, password)
            server.sendmail(sender, recipients, msg.as_string())
        print(f"  ✅ Email sent to {recipients}")
        return True
    except Exception as e:
        print(f"  ❌ Failed: {e}")
        return False

            time.sleep(1)  # rate limiting

    print(f"\n  📧 {sent} emails {'queued (dry run)' if dry_run else 'sent'}")
    if failed: print(f"  ❌ {failed} failed")

bulk_email(RECIPIENTS_CSV, dry_run=True)

import argparse, logging, sys
from pathlib import Path

def setup_logging(verbose: bool) -> None:
    level  = logging.DEBUG if verbose else logging.INFO
    format = "%(asctime)s %(levelname)-8s %(message)s" if verbose else "%(message)s"
    logging.basicConfig(level=level, format=format, datefmt="%H:%M:%S")

def cmd_convert(args: argparse.Namespace) -> int:
    log = logging.getLogger(__name__)
    log.info(f"Converting {args.input} → {args.format}")
    src = Path(args.input)
    if not src.exists():
        log.error(f"File not found: {args.input}"); return 1
    log.debug(f"Source size: {src.stat().st_size:,} bytes")
    out = src.with_suffix(f".{args.format}")
    log.info(f"✅ Would write: {out}")
    return 0

def cmd_analyse(args: argparse.Namespace) -> int:
    log = logging.getLogger(__name__)
    log.info(f"Analysing directory: {args.path}")
    p = Path(args.path)
    if not p.is_dir():
        log.error(f"Not a directory: {args.path}"); return 1
    files = list(p.rglob("*") if args.recursive else p.glob("*"))
    files = [f for f in files if f.is_file()]
    total = sum(f.stat().st_size for f in files)
    log.info(f"  Files: {len(files):,}")
    log.info(f"  Total size: {total/1_048_576:.2f} MB")
    ext_counts: dict[str, int] = {}
    for f in files: ext_counts[f.suffix.lower()] = ext_counts.get(f.suffix.lower(), 0) + 1
    for ext, n in sorted(ext_counts.items(), key=lambda x: -x[1])[:5]:
        log.info(f"  {ext or '(no ext)':<10}: {n} files")
    return 0

def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="filetool",
        description="File automation toolkit",
        formatter_class=argparse.RawDescriptionHelpFormatter,
    )
    parser.add_argument("--verbose", "-v", action="store_true", help="Debug output")
    parser.add_argument("--version", action="version",      version="%(prog)s 1.0.0")
    sub = parser.add_subparsers(dest="command", metavar="command")

    p_conv = sub.add_parser("convert", help="Convert file format")
    p_conv.add_argument("input",  help="Source file path")
    p_conv.add_argument("--format", "-f", default="json", choices=["json","csv","txt"])
    p_conv.set_defaults(func=cmd_convert)

    p_anal = sub.add_parser("analyse", help="Analyse directory contents")
    p_anal.add_argument("path",      help="Directory to analyse")
    p_anal.add_argument("--recursive", "-r", action="store_true")
    p_anal.set_defaults(func=cmd_analyse)

    return parser

def main(argv=None) -> int:
    parser = build_parser()
    args   = parser.parse_args(argv)
    setup_logging(getattr(args, "verbose", False))
    if not args.command:
        parser.print_help(); return 0
    return args.func(args)

print("Demo: filetool analyse . --recursive")
main(["analyse", ".", "--recursive"])
💡 The set_defaults(func=cmd_function) pattern makes subcommand dispatch clean — no if/elif chains needed. Add this script to your PATH as filetool and it works like any system command.

import psutil, platform
from datetime import datetime, timedelta

THRESHOLDS = {"cpu": 80, "memory": 85, "disk": 90}

def health_report() -> dict:
    cpu   = psutil.cpu_percent(interval=1)
    mem   = psutil.virtual_memory()
    disk  = psutil.disk_usage("/")
    boot  = datetime.fromtimestamp(psutil.boot_time())
    uptime = datetime.now() - boot
    procs = sorted(psutil.process_iter(["pid","name","cpu_percent","memory_percent"]),
                   key=lambda p: p.info["cpu_percent"] or 0, reverse=True)[:5]

    alerts = []
    if cpu   > THRESHOLDS["cpu"]:    alerts.append(f"🔴 CPU {cpu:.1f}% (threshold {THRESHOLDS['cpu']}%)")
    if mem.percent > THRESHOLDS["memory"]: alerts.append(f"🔴 Memory {mem.percent:.1f}%")
    if disk.percent > THRESHOLDS["disk"]:   alerts.append(f"🔴 Disk {disk.percent:.1f}%")

    status = "⚠️  ALERTS" if alerts else "✅ HEALTHY"
    print(f"\n{'═'*50}")
    print(f"  🖥️  SYSTEM HEALTH — {status}")
    print(f"  {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    print(f"{'─'*50}")
    print(f"  OS:      {platform.system()} {platform.release()}")
    print(f"  Uptime:  {str(uptime).split('.')[0]}")
    print(f"  CPU:     {cpu:.1f}% | {psutil.cpu_count()} cores")
    print(f"  Memory:  {mem.percent:.1f}% used ({mem.used/2**30:.1f}/{mem.total/2**30:.1f} GB)")
    print(f"  Disk:    {disk.percent:.1f}% used ({disk.used/2**30:.1f}/{disk.total/2**30:.1f} GB)")

    net = psutil.net_io_counters()
    print(f"  Network: ↑ {net.bytes_sent/2**20:.1f} MB sent | ↓ {net.bytes_recv/2**20:.1f} MB recv")

    if alerts:
        print(f"\n  {'─'*40}")
        for a in alerts: print(f"  {a}")

    print(f"\n  Top Processes by CPU:")
    print(f"  {'PID':<7} {'Name':<25} {'CPU%':>6} {'Mem%':>6}")
    for p in procs:
        try:
            print(f"  {p.info['pid']:<7} {p.info['name']:<25} {p.info['cpu_percent'] or 0:>5.1f}% {p.info['memory_percent'] or 0:>5.1f}%")
        except (psutil.NoSuchProcess, psutil.AccessDenied): pass

    return {"cpu": cpu, "memory": mem.percent, "disk": disk.percent, "alerts": alerts}

Run Shell Commands Safely

Execute system commands from Python with proper error handling, timeout, and output capture.

subprocessshelltimeoutcapture
Intermediate+
import subprocess, shlex
from dataclasses import dataclass

@dataclass
class CmdResult:
    returncode: int
    stdout:     str
    stderr:     str
    success:    bool

def run(
    cmd: str | list[str],
    timeout: int = 30,
    cwd: str | None = None,
    env: dict | None = None,
    shell: bool = False,
) -> CmdResult:
    if isinstance(cmd, str) and not shell:
        cmd = shlex.split(cmd)
    try:
        result = subprocess.run(
            cmd, capture_output=True, text=True,
            timeout=timeout, cwd=cwd, env=env, shell=shell
        )
        return CmdResult(result.returncode, result.stdout.strip(),
                         result.stderr.strip(), result.returncode == 0)
    except subprocess.TimeoutExpired:
        return CmdResult(-1, "", f"Timed out after {timeout}s", False)
    except FileNotFoundError:
        cmd_name = cmd[0] if isinstance(cmd, list) else cmd.split()[0]
        return CmdResult(-1, "", f"Command not found: {cmd_name}", False)

def run_pipeline(commands: list[str]) -> bool:
    for cmd in commands:
        print(f"  $ {cmd}")
        result = run(cmd)
        if result.stdout: print(f"    {result.stdout[:100]}")
        if not result.success:
            print(f"  ❌ Failed: {result.stderr}")
            return False
        print(f"  ✅ OK")
    return True

SAFE_COMMANDS = ["python --version", "echo Hello from Python automation"]
run_pipeline(SAFE_COMMANDS)
⚠️ Never pass user input directly into subprocess.run(shell=True) — this is a shell injection vulnerability. Always use shell=False (the default) and pass commands as lists. If you must use shell=True, sanitise every input.

from reportlab.lib.pagesizes  import A4
from reportlab.lib.styles    import getSampleStyleSheet, ParagraphStyle
from reportlab.lib           import colors
from reportlab.platypus      import (SimpleDocTemplate, Paragraph, Spacer,
                                       Table, TableStyle, HRFlowable)
from datetime                import datetime

def generate_report(filename: str, data: list[dict]) -> None:
    doc    = SimpleDocTemplate(filename, pagesize=A4,
                               leftMargin=50, rightMargin=50,
                               topMargin=60, bottomMargin=40)
    styles = getSampleStyleSheet()
    story  = []

    title_style = ParagraphStyle("title", parent=styles["Title"],
                                 textColor=colors.HexColor("#4B8BBE"), fontSize=18)
    story.append(Paragraph("📊 Monthly Sales Report", title_style))
    story.append(Paragraph(f"Generated: {datetime.now().strftime('%B %d, %Y')}", styles["Normal"]))
    story.append(Spacer(1, 20))
    story.append(HRFlowable(width="100%", thickness=1, color=colors.HexColor("#4B8BBE")))
    story.append(Spacer(1, 16))

    story.append(Paragraph("Summary", styles["Heading2"]))
    total = sum(r["revenue"] for r in data)
    units = sum(r["units"] for r in data)
    story.append(Paragraph(
        f"Total revenue: ₱{total:,.2f} | Total units sold: {units:,} | Regions: {len(data)}",
        styles["Normal"]
    ))
    story.append(Spacer(1, 16))

    story.append(Paragraph("Regional Breakdown", styles["Heading2"]))
    table_data = [["Region", "Units", "Revenue (₱)", "% of Total"]]
    for row in sorted(data, key=lambda r: -r["revenue"]):
        pct = row["revenue"] / total * 100
        table_data.append([row["region"], f"{row['units']:,}",
                           f"₱{row['revenue']:,.2f}", f"{pct:.1f}%"])

    tbl = Table(table_data, colWidths=[130, 80, 130, 90])
    tbl.setStyle(TableStyle([
        ("BACKGROUND",  (0,0), (-1,0), colors.HexColor("#4B8BBE")),
        ("TEXTCOLOR",   (0,0), (-1,0), colors.white),
        ("FONTNAME",    (0,0), (-1,0), "Helvetica-Bold"),
        ("ROWBACKGROUNDS", (0,1), (-1,-1), [colors.white, colors.HexColor("#F0F6FF")]),
        ("GRID",        (0,0), (-1,-1), 0.5, colors.HexColor("#CCDDEE")),
        ("ALIGN",       (1,0), (-1,-1), "RIGHT"),
    ]))
    story.append(tbl)
    story.append(Spacer(1, 20))
    story.append(Paragraph("Automated report generated by Poetic Bytes automation pipeline.",
                           styles["Italic"]))

    doc.build(story)
    print(f"  ✅ Report saved: {filename}")

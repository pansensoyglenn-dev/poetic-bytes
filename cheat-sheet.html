@property / classmethod / staticmethod
class Circle:
    def __init__(self, r): self._r = r

    @property
    def area(self):
        return 3.14159 * self._r ** 2

    @area.setter
    def area(self, v): self._r = (v/3.14)**.5

    @classmethod
    def from_diameter(cls, d): return cls(d/2)

    @staticmethod
    def validate_r(r): return r > 0

Abstract Base Classes

Advanced+
ABC & @abstractmethod
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self) -> float: ...

    @abstractmethod
    def perimeter(self) -> float: ...

class Square(Shape):
    def __init__(self, s): self.s = s
    def area(self): return self.s**2
    def perimeter(self): return 4*self.s
Protocol (structural subtyping)
from typing import Protocol

class Drawable(Protocol):
    def draw(self) -> None: ...

def render(item: Drawable) -> None:
    item.draw()

# Any class with .draw() satisfies Drawable
# without explicit inheritance — duck typing + types

Custom Iterator
class Counter:
    def __init__(self, stop):
        self.n = 0; self.stop = stop

    def __iter__(self): return self

    def __next__(self):
        if self.n >= self.stop:
            raise StopIteration
        self.n += 1
        return self.n

Decorators

Advanced+
Basic Decorator
from functools import wraps

def log_call(func):
    @wraps(func)        # preserves metadata
    def wrapper(*args, **kwargs):
        print(f"calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"done")
        return result
    return wrapper

@log_call
def add(a, b): return a + b
Parametrised Decorator
def retry(times=3):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for i in range(times):
                try: return func(*args, **kwargs)
                except Exception: pass
            raise RuntimeError("All retries failed")
        return wrapper
    return decorator

@retry(times=5)
def fetch_data(): ...

Dataclasses

Intermediate+
Basic Dataclass
from dataclasses import dataclass, field

@dataclass
class Point:
    x: float
    y: float
    label: str = ""
    tags: list[str] = field(default_factory=list)

    def distance(self) -> float:
        return (self.x**2 + self.y**2)**.5

p = Point(3.0, 4.0)
print(p)       # Point(x=3.0, y=4.0, label='')
Frozen & Slots
@dataclass(frozen=True) # immutable class Config: host: str = "localhost" port: int = 8080 @dataclass(slots=True) # faster, less RAM class Record: id: int value: float @dataclass(order=True) # enables <, >, <=, >= class Version: major: int; minor: int
__post_init__ & field()
from dataclasses import field

@dataclass
class User:
    name: str
    email: str
    _id: int = field(init=False, repr=False)

    def __post_init__(self):
        self._id = hash(self.email)
        if "@" not in self.email:
            raise ValueError("Bad email")

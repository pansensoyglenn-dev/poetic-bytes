🔒 Encapsulation with @property

Controlling access to attributes with getters and setters.

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self._balance = balance  # Protected attribute

    @property
    def balance(self):
        return self._balance

    @balance.setter
    def balance(self, value):
        if value < 0:
            raise ValueError("Balance cannot be negative")
        self._balance = value

    def deposit(self, amount):
        if amount <= 0:
            raise ValueError("Amount must be positive")
        self._balance += amount

    def withdraw(self, amount):
        if amount > self._balance:
            raise ValueError("Insufficient funds")
        self._balance -= amount

# Usage
account = BankAccount("Alice", 1000)
print(f"Balance: ${account.balance}")
account.deposit(500)
print(f"After deposit: ${account.balance}")
account.withdraw(200)
print(f"After withdrawal: ${account.balance}")
Balance: $1000
After deposit: $1500
After withdrawal: $1300
💡 @property lets you use methods like attributes — adding logic without changing the API.

📊 Fibonacci Sequence

Generate the Fibonacci sequence up to n terms.

def fibonacci_iter(n):
    """Generate Fibonacci sequence up to n terms (iterative)."""
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
    
    seq = [0, 1]
    for i in range(2, n):
        seq.append(seq[-1] + seq[-2])
    return seq

# Fibonacci with generator (memory efficient)
def fibonacci_gen(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

# Test
print("Fibonacci (iterative):", fibonacci_iter(10))
print("Fibonacci (generator):", list(fibonacci_gen(10)))

# Find the 50th Fibonacci number
fibo = fibonacci_iter(50)
print("50th Fibonacci:", fibo[-1])

🔄 GCD and LCM

Greatest Common Divisor and Least Common Multiple.

import math

a, b = 48, 18

# Using math module
print("GCD (math):", math.gcd(a, b))
print("LCM (math):", abs(a * b) // math.gcd(a, b))

# Manual Euclidean algorithm
def gcd_euclidean(a, b):
    while b:
        a, b = b, a % b
    return a

def lcm(a, b):
    return abs(a * b) // gcd_euclidean(a, b)

print("GCD (Euclidean):", gcd_euclidean(a, b))
print("LCM:", lcm(a, b))

# GCD of multiple numbers
numbers = [24, 36, 48, 60]
result = numbers[0]
for n in numbers[1:]:
    result = math.gcd(result, n)
print("GCD of", numbers, "=", result)

import functools, time

def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f} seconds")
        return result
    return wrapper

def log(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__} with args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} returned {result}")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(0.5)
    return "Done"

@log
@timer
def add(a, b):
    return a + b

print(slow_function())
print(add(3, 4))

📦 Generator Example

Memory‑efficient iteration with yield.

def squares(n):
    for i in range(n):
        yield i ** 2

# Using generator
for sq in squares(5):
    print(sq, end=" ")
print()

# List comprehension vs generator expression
list_comp = [x**2 for x in range(5)]
gen_exp = (x**2 for x in range(5))

print("List comprehension:", list_comp)
print("Generator expression:", list(gen_exp))

# Read large file line by line (memory efficient)
def read_large_file(filename):
    with open(filename, "r", encoding="utf-8") as f:
        for line in f:
            yield line.strip()

# Example (would work with any large file)
for line in read_large_file("sample.txt"):
    print("Line:", line)

# Basic comprehension

evens = [x for x in range(20) if x % 2 == 0]

matrix = [[i * j for j in range(1, 4)] for i in range(1, 4)]

nested = [[1, 2], [3, 4, 5], [6, 7]]

🧵 Exception Handling

Graceful error handling with try/except/finally.

def safe_divide(a, b):
    try:
        result = a / b
    except ZeroDivisionError:
        return "Cannot divide by zero"
    except TypeError:
        return "Invalid types provided"
    except Exception as e:
        return f"Unexpected error: {e}"
    else:
        return result
    finally:
        print("Division attempted")

print(safe_divide(10, 2))
print(safe_divide(10, 0))
print(safe_divide(10, "2"))

# Custom exception
class ValidationError(Exception):
    pass

def validate_age(age):
    if age < 0:
        raise ValidationError("Age cannot be negative")
    if age > 150:
        raise ValidationError("Age seems unrealistic")
    return "Valid age"

try:
    print(validate_age(25))
    print(validate_age(-5))
except ValidationError as e:
    print("Error:", e)

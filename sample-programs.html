║  9. Tan         10. Log      ║
║  0. Quit                     ║
╚══════════════════════════════╝""")

def get_num(prompt: str) -> float:
    while True:
        try:
            return float(input(prompt))
        except ValueError:
            print("  ❌ Invalid number. Try again.")

def calculate(choice: str) -> None:
    if choice in ("1","2","3","4","5"):
        a = get_num("  Enter first number: ")
        b = get_num("  Enter second number: ")
        ops = {
            "1": (a + b, f"{a} + {b}"),
            "2": (a - b, f"{a} - {b}"),
            "3": (a * b, f"{a} × {b}"),
            "4": (None, f"{a} ÷ {b}"),
            "5": (a ** b, f"{a}^{b}"),
        }
        if choice == "4":
            if b == 0: print("  ❌ Division by zero"); return
            result, expr = a / b, f"{a} ÷ {b}"
        else:
            result, expr = ops[choice]
        print(f"  ✅ {expr} = {result}")
    elif choice == "6":
        a = get_num("  Enter number: ")
        if a < 0: print("  ❌ Cannot sqrt negative"); return
        print(f"  ✅ √{a} = {math.sqrt(a):.6f}")
    elif choice in ("7","8","9"):
        a = get_num("  Enter angle (degrees): ")
        rad = math.radians(a)
        fns = {"7": math.sin, "8": math.cos, "9": math.tan}
        names = {"7": "sin", "8": "cos", "9": "tan"}
        print(f"  ✅ {names[choice]}({a}°) = {fns[choice](rad):.6f}")
    elif choice == "10":
        a = get_num("  Enter number: ")
        b = get_num("  Enter base (10 for log10, e for ln → enter 0): ")
        if a <= 0: print("  ❌ log requires positive number"); return
        result = math.log10(a) if b == 0 else math.log(a, b)
        print(f"  ✅ log = {result:.6f}")

def main():
    print("Welcome to Scientific Calculator")
    while True:
        show_menu()
        choice = input("  Select option: ").strip()
        if choice == "0":
            print("  Goodbye! 👋"); break
        elif choice in [str(i) for i in range(1, 11)]:
            calculate(choice)
        else:
            print("  ❌ Invalid option")
        print()

if __name__ == "__main__":
    main()

import random

LEVELS = {
    "easy":   (1, 50,  10, 1),
    "medium": (1, 100, 7,  2),
    "hard":   (1, 200, 5,  3),
}

def get_hint(guess: int, target: int, lo: int, hi: int) -> str:
    diff = abs(guess - target)
    direction = "📈 Too high!" if guess > target else "📉 Too low!"
    rng = hi - lo
    warmth = (
        "🔥 Very hot!" if diff < rng*0.05 else
        "♨️  Hot!"      if diff < rng*0.15 else
        "😐 Warm"      if diff < rng*0.30 else
        "❄️  Cold"      if diff < rng*0.50 else
        "🧊 Very cold!"
    )
    return f"  {direction}  {warmth}"

def play_round(level: str) -> int:
    lo, hi, lives, multiplier = LEVELS[level]
    target = random.randint(lo, hi)
    guesses_left = lives
    guesses_made = []

    print(f"\n🎯 I'm thinking of a number between {lo} and {hi}.")
    print(f"   You have {lives} guesses. Good luck!\n")

    while guesses_left > 0:
        print(f"  Lives: {'❤️ ' * guesses_left}  |  Guesses: {guesses_made}")
        try:
            guess = int(input(f"  Your guess ({lo}-{hi}): "))
        except ValueError:
            print("  Please enter a whole number."); continue

        if not lo <= guess <= hi:
            print(f"  Out of range ({lo}-{hi})"); continue

        guesses_made.append(guess)
        if guess == target:
            score = (lives - len(guesses_made) + 1) * 100 * multiplier
            print(f"\n  🎉 Correct! {target} in {len(guesses_made)} guess(es)!")
            print(f"  Score: +{score} points\n")
            return score

        guesses_left -= 1
        print(get_hint(guess, target, lo, hi))
        if guesses_left == 0:
            print(f"\n  💀 Game over! The number was {target}.\n")
            return 0
    return 0

def main():
    print("🎲 NUMBER GUESSING GAME")
    total_score = 0
    while True:
        print(f"\n  Total Score: {total_score} pts")
        print("  Levels: easy | medium | hard | quit")
        choice = input("  Choose: ").strip().lower()
        if choice == "quit":
            print(f"\n  Final Score: {total_score} 🏆"); break
        if choice in LEVELS:
            total_score += play_round(choice)
        else:
            print("  Unknown level")

import secrets, string, re

def generate_password(
    length: int = 16,
    use_upper: bool = True,
    use_lower: bool = True,
    use_digits: bool = True,
    use_symbols: bool = True,
    exclude: str = ""
) -> str:
    pool = ""
    guaranteed = []

    if use_upper:
        chars = "".join(c for c in string.ascii_uppercase if c not in exclude)
        pool += chars
        if chars: guaranteed.append(secrets.choice(chars))

    if use_lower:
        chars = "".join(c for c in string.ascii_lowercase if c not in exclude)
        pool += chars
        if chars: guaranteed.append(secrets.choice(chars))

    if use_digits:
        chars = "".join(c for c in string.digits if c not in exclude)
        pool += chars
        if chars: guaranteed.append(secrets.choice(chars))

    if use_symbols:
        chars = "".join(c for c in "!@#$%^&*()_+-=[]{}|;:,.<>?" if c not in exclude)
        pool += chars
        if chars: guaranteed.append(secrets.choice(chars))

    if not pool:
        raise ValueError("No characters available")

    remaining = length - len(guaranteed)
    password_chars = guaranteed + [secrets.choice(pool) for _ in range(remaining)]
    secrets.SystemRandom().shuffle(password_chars)
    return "".join(password_chars)

def strength_score(pwd: str) -> tuple[int, str]:
    score = 0
    checks = [
        (len(pwd) >= 12,  "Length ≥ 12"),
        (len(pwd) >= 16,  "Length ≥ 16 (bonus)"),
        (bool(re.search(r"[A-Z]", pwd)), "Uppercase letters"),
        (bool(re.search(r"[a-z]", pwd)), "Lowercase letters"),
        (bool(re.search(r"\d", pwd)),    "Digits"),
        (bool(re.search(r"[^A-Za-z0-9]", pwd)), "Special characters"),
    ]
    for passed, label in checks:
        if passed:
            score += 1
            print(f"  ✅ {label}")
        else:
            print(f"  ❌ {label}")

    label = (
        "⚠️  Weak"     if score <= 2 else
        "😐 Fair"     if score <= 3 else
        "👍 Good"     if score <= 4 else
        "💪 Strong"   if score <= 5 else
        "🔒 Excellent"
    )
    return score, label

def main():
    print("🔐 PASSWORD GENERATOR\n")
    try:
        length  = int(input("Password length [16]: ") or "16")
        upper   = input("Include uppercase? [Y/n]: ").lower() != "n"
        lower   = input("Include lowercase? [Y/n]: ").lower() != "n"
        digits  = input("Include digits? [Y/n]: ").lower() != "n"
        symbols = input("Include symbols? [Y/n]: ").lower() != "n"
        count   = int(input("How many passwords? [5]: ") or "5")
    except ValueError:
        print("Using defaults.")
        length, upper, lower, digits, symbols, count = 16, True, True, True, True, 5

    print(f"\n{'─'*40}")
    for i in range(1, count + 1):
        pwd = generate_password(length, upper, lower, digits, symbols)
        print(f"{i:>2}. {pwd}")

    print(f"\nStrength analysis for password #1:")
    score, label = strength_score(generate_password(length, upper, lower, digits, symbols))
    print(f"  Overall: {label}\n")

main()

import json, os
from dataclasses import dataclass, field, asdict
from datetime import datetime, date
from typing import Optional

DB_FILE = "tasks.json"

@dataclass
class Task:
    id:        int
    title:     str
    priority:  str = "medium"
    category:  str = "general"
    done:      bool = False
    due_date:  Optional[str] = None
    created:   str = field(default_factory=lambda: datetime.now().isoformat()[:10])
    notes:     str = ""

class TaskManager:
    PRIORITY_ORDER = {"high": 0, "medium": 1, "low": 2}
    PRIORITY_ICON  = {"high": "🔴", "medium": "🟡", "low": "🟢"}

    def __init__(self):
        self.tasks: list[Task] = []
        self._next_id = 1
        self._load()

    def _load(self):
        if os.path.exists(DB_FILE):
            with open(DB_FILE) as f:
                data = json.load(f)
                self.tasks = [Task(**t) for t in data["tasks"]]
                self._next_id = data.get("next_id", 1)

    def _save(self):
        with open(DB_FILE, "w") as f:
            json.dump({"tasks": [asdict(t) for t in self.tasks], "next_id": self._next_id}, f, indent=2)

    def add(self, title, priority="medium", category="general", due_date=None, notes=""):
        task = Task(self._next_id, title, priority, category, due_date=due_date, notes=notes)
        self.tasks.append(task)
        self._next_id += 1
        self._save()
        print(f"  ✅ Task #{task.id} added: '{title}'")

    def complete(self, task_id: int):
        for t in self.tasks:
            if t.id == task_id:
                t.done = True; self._save()
                print(f"  🎉 Task #{task_id} completed!"); return
        print(f"  ❌ Task #{task_id} not found")

    def delete(self, task_id: int):
        before = len(self.tasks)
        self.tasks = [t for t in self.tasks if t.id != task_id]
        if len(self.tasks) < before:
            self._save(); print(f"  🗑️  Task #{task_id} deleted")
        else:
            print(f"  ❌ Task #{task_id} not found")

    def list_tasks(self, show_done=False, category=None):
        tasks = [t for t in self.tasks
                 if (show_done or not t.done) and (category is None or t.category == category)]
        tasks.sort(key=lambda t: (self.PRIORITY_ORDER.get(t.priority, 1), t.created))
        if not tasks:
            print("  No tasks found."); return
        print(f"\n  {'ID':>3}  {'P':^3}  {'Title':<30} {'Category':<12} {'Due':>12}")
        print(f"  {'─'*3}  {'─'*3}  {'─'*30} {'─'*12} {'─'*12}")
        for t in tasks:
            icon  = self.PRIORITY_ICON.get(t.priority, "⚪")
            done  = "☑ " if t.done else "  "
            title = ("✓ " + t.title) if t.done else t.title
            due   = t.due_date or "—"
            print(f"  {t.id:>3}  {icon:^3}  {title:<30} {t.category:<12} {due:>12}")
        print(f"\n  Total: {len(tasks)} tasks | Done: {sum(1 for t in tasks if t.done)}")

    def stats(self):
        total = len(self.tasks)
        done  = sum(1 for t in self.tasks if t.done)
        cats  = {}
        for t in self.tasks: cats[t.category] = cats.get(t.category, 0) + 1
        print(f"\n  📊 Stats: {total} total | {done} done | {total-done} pending")
        print("  Categories:", " | ".join(f"{k}({v})" for k, v in cats.items()))

def main():
    mgr = TaskManager()
    print("✅ TODO MANAGER  (type 'help' for commands)")
    SAMPLE_TASKS = [
        ("Learn Python decorators",     "high",   "study"),
        ("Build portfolio project",      "high",   "career"),
        ("Read The Pragmatic Programmer", "medium", "study"),
        ("Exercise 30 minutes",           "low",    "health"),
    ]
    if not mgr.tasks:
        print("  Loading sample tasks…")
        for title, prio, cat in SAMPLE_TASKS:
            mgr.add(title, prio, cat)
    mgr.list_tasks()
    mgr.stats()
    print("\n  Commands: list | done  | delete  | add | quit")
    while True:
        cmd = input("\n  > ").strip().lower().split()
        if not cmd: continue
        if cmd[0] == "quit": print("  Saved. Goodbye!"); break
        elif cmd[0] == "list": mgr.list_tasks(show_done="all" in cmd)
        elif cmd[0] == "done" and len(cmd) > 1: mgr.complete(int(cmd[1]))
        elif cmd[0] == "delete" and len(cmd) > 1: mgr.delete(int(cmd[1]))
        elif cmd[0] == "stats": mgr.stats()
        elif cmd[0] == "add":
            title = input("  Title: ")
            prio  = input("  Priority [high/medium/low]: ") or "medium"
            cat   = input("  Category: ") or "general"
            due   = input("  Due date (YYYY-MM-DD, blank to skip): ") or None
            mgr.add(title, prio, cat, due)
        else: print("  Commands: list | list all | done  | delete  | add | stats | quit")

import re, statistics
from collections import Counter

def analyse_text(text: str) -> dict:
    words_raw  = re.findall(r"\b[a-zA-Z']+\b", text.lower())
    sentences  = re.split(r"[.!?]+", text.strip())
    sentences  = [s.strip() for s in sentences if s.strip()]
    paragraphs = [p.strip() for p in text.split("\n\n") if p.strip()]

    STOP_WORDS = {
        "the","a","an","and","or","but","in","on","at",
        "to","for","of","is","it","was","be","with","as",
        "by","from","this","that","are","were","he","she",
        "they","we","i","you","not","have","has","had"
    }
    keywords = [w for w in words_raw if w not in STOP_WORDS and len(w) > 2]
    word_freq = Counter(words_raw)
    kw_freq   = Counter(keywords)

    word_lengths = [len(w) for w in words_raw]
    sent_lengths = [len(re.findall(r"\b\w+\b", s)) for s in sentences]

    syllables = sum(
        max(1, len(re.findall(r"[aeiou]", w, re.I)) - 1 * (w.endswith("e") and len(w) > 3))
        for w in words_raw
    )
    fk_grade = (
        0.39 * (len(words_raw) / max(1, len(sentences))) +
        11.8 * (syllables / max(1, len(words_raw))) - 15.59
    )

    return {
        "char_count":    len(text),
        "word_count":    len(words_raw),
        "unique_words":  len(word_freq),
        "sentence_count":len(sentences),
        "paragraph_count":len(paragraphs),
        "avg_word_len":  statistics.mean(word_lengths) if word_lengths else 0,
        "avg_sent_len":  statistics.mean(sent_lengths) if sent_lengths else 0,
        "reading_time_min": round(len(words_raw) / 200, 1),
        "top_words":     word_freq.most_common(10),
        "top_keywords":  kw_freq.most_common(8),
        "fk_grade":      round(fk_grade, 1),
        "vocabulary_richness": round(len(word_freq) / max(1, len(words_raw)) * 100, 1),
    }

def print_report(stats: dict) -> None:
    print("\n" + "═"*50)
    print("   📊 TEXT ANALYSIS REPORT")
    print("═"*50)
    print(f"  Characters:        {stats['char_count']:>8,}")
    print(f"  Words:             {stats['word_count']:>8,}")
    print(f"  Unique words:      {stats['unique_words']:>8,}")
    print(f"  Sentences:         {stats['sentence_count']:>8,}")
    print(f"  Paragraphs:        {stats['paragraph_count']:>8,}")
    print(f"  Avg word length:   {stats['avg_word_len']:>8.1f} chars")
    print(f"  Avg sentence:      {stats['avg_sent_len']:>8.1f} words")
    print(f"  Reading time:      {stats['reading_time_min']:>8} min")
    print(f"  Vocabulary richness: {stats['vocabulary_richness']:>6.1f}%")
    fk = stats["fk_grade"]
    fk_label = ("Very Easy" if fk < 6 else "Easy" if fk < 8 else
                "Standard" if fk < 10 else "Difficult" if fk < 12 else "Very Difficult")
    print(f"  Flesch-Kincaid:    {fk:>5} ({fk_label})")
    print("\n  Top 10 Words:")
    for word, count in stats["top_words"]:
        bar = "█" * min(count, 20)
        print(f"    {word:<15} {bar} {count}")
    print("\n  Top Keywords (excl. stop words):")
    print("  " + " · ".join(f"{w}({c})" for w, c in stats["top_keywords"]))

import csv, io, re
from dataclasses import dataclass, field

@dataclass
class Contact:
    name:    str
    phone:   str = ""
    email:   str = ""
    group:   str = "general"
    address: str = ""
    notes:   str = ""

    def matches(self, query: str) -> bool:
        q = query.lower()
        return any(q in f.lower() for f in [self.name, self.phone, self.email, self.group])

class ContactBook:
    def __init__(self): self._contacts: list[Contact] = []

    def add(self, c: Contact):
        self._contacts.append(c)
        print(f"  ✅ Added: {c.name}")

    def search(self, query: str) -> list[Contact]:
        return [c for c in self._contacts if c.matches(query)]

    def delete(self, name: str) -> bool:
        before = len(self._contacts)
        self._contacts = [c for c in self._contacts if c.name.lower() != name.lower()]
        return len(self._contacts) < before

    def list_all(self, group=None):
        contacts = [c for c in self._contacts if not group or c.group == group]
        contacts.sort(key=lambda c: c.name.lower())
        print(f"\n  {'Name':<20} {'Phone':<15} {'Email':<25} {'Group'}")
        print(f"  {'─'*20} {'─'*15} {'─'*25} {'─'*10}")
        for c in contacts:
            print(f"  {c.name:<20} {c.phone:<15} {c.email:<25} {c.group}")
        print(f"\n  {len(contacts)} contact(s)")

    def export_csv(self) -> str:
        buf = io.StringIO()
        w = csv.DictWriter(buf, fieldnames=["name","phone","email","group","address","notes"])
        w.writeheader()
        for c in self._contacts: w.writerow({"name":c.name,"phone":c.phone,"email":c.email,"group":c.group,"address":c.address,"notes":c.notes})
        return buf.getvalue()

    def groups(self) -> dict:
        g = {}
        for c in self._contacts: g[c.group] = g.get(c.group, 0) + 1
        return g

def main():
    book = ContactBook()
    SAMPLES = [
        Contact("Ana García",    "+63-912-345-6789", "ana@email.com",    "friends"),
        Contact("Bob Chen",       "+1-555-234-5678",  "bob@work.com",     "work"),
        Contact("Carol Smith",    "+44-20-1234-5678", "carol@uni.edu",    "school"),
        Contact("David Nguyen",   "+63-917-456-7890", "david@email.com",  "friends"),
        Contact("Eve Patel",      "+91-98-1234-5678", "eve@startup.io",   "work"),
    ]
    for c in SAMPLES: book.add(c)
    book.list_all()
    print("\n  Groups:", book.groups())

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional

@dataclass
class Transaction:
    amount:      float
    tx_type:     str
    description: str
    balance:     float
    timestamp:   str = field(default_factory=lambda: datetime.now().strftime("%Y-%m-%d %H:%M"))

class InsufficientFundsError(Exception): pass

class BankAccount(ABC):
    _next_id = 1000

    def __init__(self, owner: str, initial_deposit: float = 0):
        BankAccount._next_id += 1
        self.account_id  = f"ACC-{BankAccount._next_id}"
        self.owner       = owner
        self._balance    = 0.0
        self._history: list[Transaction] = []
        if initial_deposit > 0: self.deposit(initial_deposit, "Account opening")

    @property
    def balance(self) -> float: return self._balance

    @abstractmethod
    def account_type(self) -> str: ...

    def deposit(self, amount: float, desc: str = "Deposit") -> None:
        if amount <= 0: raise ValueError("Amount must be positive")
        self._balance += amount
        self._history.append(Transaction(amount, "credit", desc, self._balance))
        print(f"  ✅ +₱{amount:,.2f} → Balance: ₱{self._balance:,.2f}")

    def withdraw(self, amount: float, desc: str = "Withdrawal") -> None:
        if amount <= 0: raise ValueError("Amount must be positive")
        self._check_withdrawal(amount)
        self._balance -= amount
        self._history.append(Transaction(amount, "debit", desc, self._balance))
        print(f"  ✅ -₱{amount:,.2f} → Balance: ₱{self._balance:,.2f}")

    def _check_withdrawal(self, amount: float) -> None:
        if amount > self._balance:
            raise InsufficientFundsError(f"Balance ₱{self._balance:,.2f} < ₱{amount:,.2f}")

    def transfer(self, target: "BankAccount", amount: float) -> None:
        self.withdraw(amount, f"Transfer to {target.account_id}")
        target.deposit(amount, f"Transfer from {self.account_id}")

    def statement(self, last_n: int = 10) -> None:
        print(f"\n  ┌{'─'*55}┐")
        print(f"  │  {self.account_type():^51}  │")
        print(f"  │  {self.account_id} — {self.owner:<30} │")
        print(f"  │  Balance: ₱{self.balance:>10,.2f}{' '*26} │")
        print(f"  └{'─'*55}┘")
        print(f"  {'Date':<17} {'Type':<8} {'Description':<22} {'Amount':>10}")
        print(f"  {'─'*17} {'─'*8} {'─'*22} {'─'*10}")
        for tx in self._history[-last_n:]:
            sign = "+" if tx.tx_type == "credit" else "-"
            icon = "💚" if tx.tx_type == "credit" else "🔴"
            print(f"  {tx.timestamp:<17} {icon:<8} {tx.description:<22} ₱{sign}{tx.amount:>9,.2f}")

class SavingsAccount(BankAccount):
    def __init__(self, owner, initial=0, interest_rate=0.035):
        self._rate = interest_rate
        super().__init__(owner, initial)
    def account_type(self): return "SAVINGS ACCOUNT"
    def apply_interest(self):
        interest = self._balance * self._rate / 12
        self.deposit(interest, "Monthly interest")
        print(f"  💸 Interest applied: ₱{interest:,.2f} ({self._rate*100:.1f}% p.a.)")

class CheckingAccount(BankAccount):
    def __init__(self, owner, initial=0, overdraft_limit=5000):
        self._overdraft = overdraft_limit
        super().__init__(owner, initial)
    def account_type(self): return "CHECKING ACCOUNT"
    def _check_withdrawal(self, amount):
        if amount > self._balance + self._overdraft:
            raise InsufficientFundsError(f"Exceeds balance + overdraft limit of ₱{self._overdraft:,.0f}")

@dataclass
class Product:
    sku:         str
    name:        str
    category:    str
    price:       float
    cost:        float
    quantity:    int
    reorder_at:  int = 10

    @property
    def margin(self) -> float:

    @property
    def stock_value(self) -> float:

@dataclass
class SaleRecord:
    sku:       str
    qty:       int
    revenue:   float
    profit:    float
    timestamp: str = field(default_factory=lambda: datetime.now().strftime("%Y-%m-%d %H:%M"))

class Inventory:
    def __init__(self, store_name: str):
        self.store_name = store_name
        self._products: dict[str, Product] = {}
        self._sales: list[SaleRecord] = []

    def add_product(self, p: Product) -> None:
        self._products[p.sku] = p
        print(f"  ✅ Added: [{p.sku}] {p.name} — ₱{p.price:,.2f} x {p.quantity} units")

    def restock(self, sku: str, qty: int) -> None:
        if sku not in self._products:
            print(f"  ❌ SKU not found: {sku}"); return
        self._products[sku].quantity += qty
        print(f"  📦 Restocked [{sku}]: +{qty} units → {self._products[sku].quantity} total")

    def sell(self, sku: str, qty: int) -> bool:
        if sku not in self._products:
            print(f"  ❌ SKU not found: {sku}"); return False
        p = self._products[sku]
        if p.quantity < qty:
            print(f"  ❌ Insufficient stock. Have {p.quantity}, need {qty}"); return False
        p.quantity -= qty
        revenue = p.price * qty
        profit  = (p.price - p.cost) * qty
        self._sales.append(SaleRecord(sku, qty, revenue, profit))
        print(f"  💰 Sold {qty}x {p.name} — Revenue: ₱{revenue:,.2f} | Profit: ₱{profit:,.2f}")
        if p.is_low_stock:
            print(f"  ⚠️  LOW STOCK: [{sku}] {p.name} — only {p.quantity} left!")
        return True

Event Scheduler with Priority Queue

Schedule tasks with recurrence rules, priorities, heapq-based dispatch, and conflict detection.

import heapq
from dataclasses import dataclass, field
from datetime import datetime, timedelta
from enum import IntEnum

class Priority(IntEnum):
    CRITICAL = 1
    HIGH     = 2
    MEDIUM   = 3
    LOW      = 4

PRIORITY_ICON = {
    Priority.CRITICAL: "🔴",
    Priority.HIGH:     "🟠",
    Priority.MEDIUM:   "🟡",
    Priority.LOW:      "🟢",
}

@dataclass(order=True)
class ScheduledEvent:
    scheduled_at: datetime
    priority:     Priority
    title:        str = field(compare=False)
    duration_min: int = field(compare=False, default=60)
    description:  str = field(compare=False, default="")
    recur_days:   int | None = field(compare=False, default=None)
    completed:    bool = field(compare=False, default=False)

    @property
    def ends_at(self) -> datetime:
        return self.scheduled_at + timedelta(minutes=self.duration_min)

    def overlaps_with(self, other: "ScheduledEvent") -> bool:
        return self.scheduled_at < other.ends_at and other.scheduled_at < self.ends_at

class Scheduler:
    def __init__(self):
        self._heap: list[ScheduledEvent] = []
        self._all:  list[ScheduledEvent] = []

    def schedule(self, event: ScheduledEvent) -> bool:
        conflicts = [e for e in self._all if not e.completed and e.overlaps_with(event)]
        if conflicts:
            print(f"  ⚠️  CONFLICT with: {', '.join(c.title for c in conflicts)}")
            return False
        heapq.heappush(self._heap, event)
        self._all.append(event)
        icon = PRIORITY_ICON[event.priority]
        print(f"  ✅ {icon} Scheduled: {event.title} @ {event.scheduled_at.strftime('%a %d %b %H:%M')}")
        return True

    def next_event(self) -> ScheduledEvent | None:
        while self._heap and self._heap[0].completed:
            heapq.heappop(self._heap)
        return self._heap[0] if self._heap else None

    def complete_next(self) -> None:
        event = self.next_event()
        if not event: print("  No events"); return
        event.completed = True
        heapq.heappop(self._heap)
        print(f"  🎉 Completed: {event.title}")
        if event.recur_days:
            next_e = ScheduledEvent(
                event.scheduled_at + timedelta(days=event.recur_days),
                event.priority, event.title, event.duration_min,
                event.description, event.recur_days
            )
            self.schedule(next_e)

    def agenda(self, days: int = 7) -> None:
        now    = datetime.now()
        cutoff = now + timedelta(days=days)
        upcoming = sorted([e for e in self._all
                           if not e.completed and now <= e.scheduled_at <= cutoff])
        print(f"\n  📅 AGENDA — Next {days} days ({len(upcoming)} events)")
        print(f"  {'─'*65}")
        if not upcoming: print("  No events scheduled."); return
        current_day = None
        for e in upcoming:
            day = e.scheduled_at.strftime("%A %d %B")
            if day != current_day:
                print(f"\n  📆 {day}"); current_day = day
            icon = PRIORITY_ICON[e.priority]
            start = e.scheduled_at.strftime("%H:%M")
            end   = e.ends_at.strftime("%H:%M")
            recur = f"↻ every {e.recur_days}d" if e.recur_days else ""
            print(f"  {icon} {start}–{end}  {e.title:<28} {recur}")

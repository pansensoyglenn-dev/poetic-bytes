slicing ‚Äì [start:stop:step]

list comprehensions

# squares of even numbers [x*x for x in range(10) if x % 2 == 0] # [0, 4, 16, 36, 64] # flatten matrix matrix = [[1,2],[3,4],[5,6]] [item for row in matrix for item in row] # [1,2,3,4,5,6]
